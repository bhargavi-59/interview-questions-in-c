#include <stdio.h>
void main()
{
    int small;
    int big;
    int goal;
    printf("Enter no of small bricks\n");
    scanf("%d",&small);
    printf("Enter the no of big bricks\n");
    scanf("%d",&big);
    printf("Enter the no of bricks required to reach the goal\n");
    scanf("%d",&goal);
    int req=goal/5;
    if(req<=big)
    {
        int rem=goal-(req*5);
        if(rem<=small)
        {
            printf("true");
        }
        else
        {
            printf("false");
        }
    }
        else
        {
            int rem1=goal-(big*5);
            if (rem1<=small)
            {
                printf("true");
            }
            else
            {
                printf("false");
            }
        }
}
