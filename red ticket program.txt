#include <stdio.h>
void main()
{
    int a=0;
    int b=1;
    int c=2;
    printf("Enter the value of a,b,c\n");
    scanf("%d,%d,%d",&a,&b,&c);
    if(a==2&&b==2&&c==2)
    {
        printf("10");
    }
    else if(a==b&&b==c&&c==a)
    {
        printf("5");
    }
    else if(b!=a&&c!=a)
    {
        printf("1");
    }
    else
    {
        printf("0");
    }
}
